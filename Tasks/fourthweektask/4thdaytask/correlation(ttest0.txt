import numpy as np

def pearson_correlation(x, y):
    mean_x = np.mean(x)
    mean_y = np.mean(y)
    std_x = np.std(x, ddof=1)
    std_y = np.std(y, ddof=1)
    
    covariance = np.mean((x - mean_x) * (y - mean_y))
    
    correlation_coefficient = covariance / (std_x * std_y)
    
    return correlation_coefficient
